class TestArray(unittest.TestCase): 

def test_basic1(self): 

a = Array() 

a.value = 1 

self.assertEqual(a.value, a["value"]) 

 

def test_basic2(self): 

test_arr = [0, 1, 2, 3, 4] 

a = Array(test_arr) 

self.assertEqual(a[3], 3) 

self.assertEqual(list(a._dict.values()), test_arr) 

self.assertEqual(len(a), len(test_arr)) 

self.assertEqual(a, test_arr) 

 

test_arr = [3, 4, 2, 1, 0] 

a = Array(test_arr) 

self.assertEqual(list(a._dict.values()), test_arr) 

a.append(10) 

self.assertEqual(a[5], 10) 

 

def test_repr(self): 

test_arr = [3, 4, 5, 6, 7] 

a1 = Array(test_arr) 

self.assertEqual(repr(a1), "<Array %r>" % test_arr) 

 

def test_iter(self): 

test_arr = [3, 4, 5, 6, 7] 

a1 = Array(test_arr) 

a2 = [x for x in a1] 

self.assertEqual(test_arr, a2) 

 

test_arr = [8, 7, 6, 4, 2] 

a1 = Array(test_arr) 

a2 = [x for x in a1] 

self.assertEqual(test_arr, a2)
